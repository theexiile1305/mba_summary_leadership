    \begin{sectionbox}{Bedeutung guter Führung}
        \begin{itemize}
            \item historisch \& aktuell von großer Bedeutung\\
            \ra durch Mächtigkeit \& Dynamik von Persönlichkeit
            \item kaum beschreibbar, wie man richtig führt
            \item unterschiedliche Ansätze \& Erfolgsfaktoren
            \item mehrere Variablen zur Messung des Erfolgs
        \end{itemize}

    \begin{warningbox}{Ansätze der Führung}
        \begin{itemize}
            \item \textbf{Rolle}: Verknüpft mit Erwartungen an FK
            \item \textbf{Soziale Einflussnahme}: Einwirken auf MA durch Führung, um sinnvolles Verhalten zu erreichen
        \end{itemize}
    \end{warningbox}

        \begin{itemize}

            \item Ressourcenallokation für Ziele \& Maßnahmen
            \item Ausgestaltung von Bürokratie \& Systemen
        \end{itemize}
    \end{sectionbox}

    \section{Führungstheorien im Wandel der Zeit}   
        \begin{sectionbox}{Eigenschaftstheorie - \emph{Great Man Theory}}
        \begin{itemize}
            \item Sozialdarwinistisches Elitedenken
            \item Geborene Führungspersönlichkeiten\\
            \ra Keine Trainier- oder Erlernbarkeit
            \item Ablehnung der Entscheidungspartizipation von MA
            \item ursprüngliche Eigenschaften\\
            \ra zeitstabil \& situationsabhängig\\
            \ra messbar \& klar feststellbar
            \item statisch ohne Berücksichtigung der Führungssituation
            \item Weiterentwicklung via Stogdill \& \emph{BIG FIVE}
        \end{itemize}
    \end{sectionbox}
        \begin{hintbox}{Stogdill (1948): Attribute für Führungserfolg}
            Fähigkeit, Leistung, Verantwortung, Partizipation, Status
        \end{hintbox}
        \begin{hintbox}{Persönlichkeitsskala \emph{BIG FIVE}}
            Extraversion, Verträglichkeit, Offenheit, Gewissenhaftigkeit und emotionale Stabilität
        \end{hintbox}
    \begin{sectionbox}{Führungsstilkontinuum}
        \begin{itemize}
            \item Keine Berücksichtigung von MA/Situation
            \item eindimensional von autoritär bis teil-autonom
            \item Richtiger Führungsstil in Abhängigkeit von\\
            \ra Qualifikation MA \& Arbeitsinhalt\\
            \ra Persönlichkeitsmerkmale FK \& Teamgröße\\
            \ra Akzeptanz FK im Team \& Branche
        \end{itemize}
    \end{sectionbox}
    \subsection{Berücksichtigung der Situation}
    \begin{normbox}{\subsubsection{Kontingenztheorie nach \emph{Fiedler}}}
        \begin{itemize}
            \item Führungsstil in Abhängigkeit von Situation
            \item Führungssituation bedingt durch
            \begin{itemize}
                \item \emph{KF-MA-Beziehung}\\
                \ra Wertschätzung der FK durch MA
                \item \emph{Positionsmacht}\\
                \ra Einfluss der FK infolge hierarchischer Stellung
                \item \emph{Strukturierungsgrad der Aufgabe}\\
                \ra Anzahl wiederkehrender Elemente, deren Planbarkeit \& Überprüfbarkeit der Leistungen\\
                \ra Je klarer die Aufgabe strukturiert, desto leichter fallen Koordination \& Kontrolle der MA
            \end{itemize}
            \item Erfolg dieser Führungsstile bedingt je nach Situationen\\
            \ra günstig \& ungünstig: leistungsbezogene Führung\\
            \ra mittelgünstig: mitarbeiterbezogene Führung
            \item Kritik: Laborexperiment \rightleftharpoons !Proof in Feldstudien
        \end{itemize}
    \end{normbox}
    \begin{normbox}{\subsubsection{Reifegradmodell nach \emph{Hersey \& Blanchard}}}
        \begin{itemize}
            \item Führungsverhalten abhängig von Reifegrad des MA\\
            \ra psychologisch: Motivation \& Verantwortung\\
            \ra funktional: Ausbildung \& Erfahrung
            \ra individualisierte Führung \ra kein Gießkannenprinzip
            \begin{hintbox}{$psychologische \ Reife \bullet funktionale \ Reife$}
                \ra unterweisend: gering \& gering\\
                \ra verkaufend: hoch \& gering\\
                \ra partizipierend: gering \& hoch\\
                \ra delegierend: hoch \& hoch
            \end{hintbox}
            \item Ziel: Maximierung des Reifegrads des MA
            \item Plausibilität des Modells, hohe Praxistauglichkeit 
            \item Kritik: !empirischer Beleg
        \end{itemize}
    \end{normbox}
    \begin{normbox}{\subsubsection{Entscheidungsmodell von \emph{Vroom \& Yetton}}}
        \begin{itemize}
            \item empirisch nachgewiesener Entscheidungsbaum (6Q)
            \item Entscheidungen des Führungsstil basierend auf\\
            \ra Autokratie\\
            \ra konsultativ auf Beratung der Geführten\\
            \ra Gruppe
            \item Führungsentscheidung basierend auf\\
            \ra Qualität \& Akzeptanz der Entscheidung\\
            \ra Ökonomie des Entscheidungsverhaltens
        \end{itemize}
    \end{normbox}
    \begin{sectionbox}{Systematische Führung}
        \begin{itemize}
            \item Führungshandlung in komplex vernetzten Sozialgefüge mit vielen direkten \& indirekten Reaktionen\\
            \ra Einflussnahme auf System durch Kommunikation\\
            \ra Begreifen komplexr und Bilateraler Interaktionen
            \item Systeme: alle Relationen \& Interaktionen zueinander
            \item System: Getriebe \& Unternehmen
            \item Basiselement: Kommunikation
            \item Problem: FK ist Teil des System \rightleftharpoons Einflüsse!
        \end{itemize}
    \end{sectionbox}
        \begin{sectionbox}{Symbolische Führung}
        \begin{itemize}
            \item Symbolisierung von Werten \& Überzeugungen\\
            \ra oft große Interpretation von MA
            \item aufgesetzte Symbolik häufig unauthentisch
            \item unterschiedliche Interpretation der Führungshandlung
            \item Kritik: empirisch ungenügend erforscht
        \end{itemize}
    \end{sectionbox}
    \subsection{Transaktionale \& transformationale Führung}
    \begin{normbox}{\subsubsection{Allgemeines}}
        \begin{itemize}
            \item Verbreitung seit 30y des Ansatz in Industrie/Science
            \item transaktionaler Führung \ra transformationale Führung\\
            \ra positive Verstärkung \& Vorbildfunkton der FK
        \end{itemize}
    \end{normbox}
    \begin{normbox}{\subsubsection{Transaktionale Führung}}
        \begin{itemize}
            \item Basis: lerntheoretisches Prinzip der Verstärkung
            \ra operante Konditionierung \rightleftharpoons \emph{carrots-and-sticks}
            \item FK kontrolliert Aufgaben \& Ziele des MA
            \item korrespondiert mit \emph{Management by Exception}
            \item Fairer Austausch zwischen FK \& MA
        \end{itemize}
    \end{normbox}
    \begin{normbox}{\subsubsection{Transformationale Führung}}
        \begin{itemize}
            \item Motivation der MA durch Werte \& Gefühle
            \item Ergänzung transaktionalen Führung durch\\
            \ra langfristige \& übergeordnete Werte ergänzen kurzfristige materielle Ziele
            \item fairer Austausch von Leistung \rightleftharpoons Gegenleistung
            \item Fokus: Vertrauen, Sinn, Inspiration \& Wertschätzung
        \end{itemize}
    \end{normbox}
    \begin{warningbox}{Motivation der MA durch Werte \& Gefühle auf 4 Ebenen}
        \begin{itemize}
            \item \emph{Inspirational Motivation}: überzeugende Vermittlung attraktiver Visionen
            \item \emph{Idealized Influence}: glaubwürdige Verkörperung eines Vorbildes
            \item \emph{Intellectual Stimulation}: Anregung zum unabhängigen Denken, Unterstützung von Veränderung
            \item \emph{Individualized Consideration}: durch die Unterstützung der MA-Entwicklung, individuelle Wertschätzung
        \end{itemize}
    \end{warningbox}
    \begin{hintbox}{Rollenverhalten von Führungskräften - Gegenüberstellung}
        \begin{itemize}
            \item Koordinationsmechanismen\\
            \ra Verträge, Belohnung \& Bestrafung (\emph{TA})\\
            \ra Begeisterung \& Vertrauen, Kreativität (\emph{TF})
            \item MA-Motivation: extrinsisch (\emph{TA}), intrinsisch (\emph{TF})
            \item Zielerreichung: kurzfristig (\emph{TA}), langfristig (\emph{TF})
            \item Zielinhalte: materiell (\emph{TA}), ideel (\emph{TF})
            \item Rolle der FK: Instrukteur (\emph{TA}), Lehrer \& Coach (\emph{TF})

    \end{hintbox}
    \begin{sectionbox}{Führungstheorien im Wandel der Zeit: \\ Führung im Spannungsfeld}

            \item \textbf{Basis sind menschliche Motive}\\
            \ra charakteristische Werthaltungen\\
            \ra Handlungsdispositionen als Voraussetzungen\\
            \ra Bezug zu Erlebnis, Leistung, Macht \& Aggression 
            \item \textbf{Motive als Beweggründe für Handlungen}\\
            \ra nicht beobachtbar\\
            \ra zeitlich \& über mögliche Situationen hinweg stabil
        \end{itemize}
    \end{normbox}
    \begin{warningbox}{Bedürfnisse (Motive) nach \emph{Maslow}}
        \begin{itemize}
            \item Wachstumsbedürfnis: Erfüllung unterer Bedürfnisse!
            \item Modell erforderlich! \textbf{Wieso?}\\
            \ra Einschätzung von MA\\
            \ra Geben von adäquaten Anreize
            \item \textbf{Defizitärbedürfnis}: Beseitigung eines Mangels
            \item \textbf{Wachstumsbedürfnis}: Motive zur Selbstverwirklichung
        \end{itemize}
    \end{warningbox}
    \begin{warningbox}{ERG-Theorie nach \emph{Alderfer}}
        \begin{itemize}
            \item \textbf{Weiterentwicklung} der Bedürfnisse nach Maslow
            \item \textbf{Drei Bedürfnisklassen}\\
            \ra Existenzbedürfnisse (\emph{Existence needs})\\
            \ra soziale Bedürfnisse (\emph{Relatedness needs})\\
            \ra Wachstumsbedürfnisse (\emph{Growth needs})
            \item Je \textbf{geringer Erfüllungsgrad}, desto \textbf{stärker Bedürfnisse}
        \end{itemize}
    \end{warningbox}
    \begin{normbox}{\subsubsection{Grundmodell der Motivation}}
        \textbf{Motivation als aktivierende Verhaltensbereitschaft eines Individuums in Hinblick auf Erreichen bestimmter Ziele}
        \begin{itemize}
            \item als hypothetisches Konstrukt, nicht messbar
            \item Motive via testpsychologischer Verfahren
        \end{itemize}
        \includegraphics[width=\textwidth]{media/grundmodell_motivation.png}
        \begin{itemize}
            \item \textbf{Anreize}: Bedürfnisse \& Motive situativ angesprochen
            \item \textbf{Volition (Willensakt)}: Umsetzung von Motivationstendenzen in konkrete Handlungen
        \end{itemize}
    \end{normbox}
    \begin{normbox}{\subsubsection{Intrinsische \& extrinsische Motivation}}
        \begin{itemize}
            \item \textbf{Extrinsisch}: äußere Anreize, um Ziele zu erreichen
            \item \textbf{Intrinsisch}: Tätigkeit um ihrer selbst willen ausgeübt\\
            \ra nicht aufgrund der erwarteten Ergebnisse
        \end{itemize}
    \end{normbox}
    \begin{normbox}{\subsubsection{Motivation: VIE-Theorie nach \emph{Vroom}}}
        \begin{itemize}
            \item in Abhängigkeit von Komponenten der Zielerreichung\\
            \ra Anstrengungen nur, wenn lohnend in Bezug auf Ziel
            \item Faktoren mit Auswirkung auf Motivation\\
            \ra \textbf{Valenz}: Wert eines Handlungsergebnisses im Sinne positiver/negativer affektiver Bedeutung des Ergebnis\\
            \ra \textbf{Instrumentalität}: Schätzwert für Verknüpfungsgrad zwischen Handlungsergebnis \& Handlungsfolge\\
            \ra \textbf{Erwartungen}: subjektiv wahrgenommene Wahrscheinlichkeit konkreter Handlungsausgänge
        \end{itemize}
        \emph{Handlungstendenz = Valenz d. Ergebnisses $\cdot$ Erwartungen}
    \end{normbox}
    \begin{sectionbox}{Kommunikation}
        \begin{itemize}
            \item Gespräche: wesentliche Kommunikationsform
            \item Wichtiger Zeitfaktor bei FK: adhoc vs. kurz\\
            \ra formalere Feedback- \& Beurteilungsgespräche
            \item Quantität vs. Qualität entscheidend
        \end{itemize}
    \end{sectionbox}
    \begin{warningbox}{Eisbergmodell}
        \begin{itemize}
            \item $\frac{1}{7}$ Kommunikation: bewusst
            \item $\frac{6}{7}$ Kommunikation: unbewusst \& unausgesprochen\\
            \ra Mimik, Gestik, Körperhaltung, \& Schweigen\\
            \ra Distanz zum Gesprächspartner
        \end{itemize}
    \end{warningbox}
    \begin{warningbox}{Kommunikationsquadrat nach \emph{Schulz von Thun}}
        \begin{itemize}
            \item Austausch von Informationen via Sender \& Empfänger\\
            \item Überprüfen der Verständigung über Feedback
            \item Feedback macht Kommunikation offener \& klarer
            \ra vermindert Gefahr von Missverständnissen\\
            \ra wichtig für Führungsprozess
            \item Vier Seiten einer Nachricht\\
            \ra \textbf{Sachinhalt}: Worüber informiert man?\\
            \ra \textbf{Beziehung}: Wie stehe ich zu Dir?\\
            \ra \textbf{Selbstkundgabe}: Was man von sich preis gibt!\\
            \ra \textbf{Appell}: Wozu möchte ich Dich veranlassen?
            \item \textbf{Missverständnis} \ra andere Ebene wahrgenommen
        \end{itemize}
    \end{warningbox}
    \begin{warningbox}{Transaktionsanalyse}
        \begin{itemize}
            \item Aufschluss über Missverständnissen \& Konflikten\\
            \ra bedingt durch (verdeckte) Überkreuztransaktionen
            \item basierend auf 3 unterschiedliche \emph{Ich-Zustände}
            \item \textbf{Eltern-Ich}: angelerntes Werturteil \& Besserwisserei
            \item \textbf{Kindheits-Ich}:  Gefühle/Bedürfnisse im Mittelpunkt \\
            \ra frei: Fantasie, Liebe \&  Kreativität\\
            \ra rebellisch: Aufmerksamkeit \&  !Bevormundung\\
            \ra angepasst: Nachgeben \& Vermeidung
            \item \textbf{Erwachsenen-Ich}: Faktensammlung \la\ra \textbf{FK!}\\
            \ra aktive, abwägende, sachliche \& faire Entscheidung
        \end{itemize}
    \end{warningbox}

    \begin{normbox}{\subsubsection{Allgemeines}}
        \begin{itemize}
            \item Beurteilungsgespräch = Feedbackgespräch\\

            \ra als Grundlage für Entlohnung, periodisch

            \ra FK: direkt \& Zirkel von FK\\
            \ra Experten: Coaches \& Personaler\\

        \end{itemize}
    \end{normbox}
    \begin{hintbox}{360 Grad-Feedback}
        \begin{itemize}
            \item Beurteilung von FK durch Rückmeldung
            \item Feedbackgeber: geführte MA, Kollegen \& Vorgesetzte
            \item häufig schriftlich \& anonymisiert

        \end{itemize}

            \item \textbf{Entscheidungsunterstützungsfunktion}: Vergütung, Personalentwicklung, Team-/Mitarbeiterführung

    \section{Teams}
    \begin{warningbox}{Definition Teamarbeit}
        Optimale Gestaltung arbeitsteiliger Strukturen unter Nutzung der speziellen Fähigkeiten der Teammitglieder, insbesondere bei der Bewältigung komplexer Aufgaben.
    \end{warningbox}
    \begin{sectionbox}{Arten von Teams}
        \begin{itemize}
            \item \textbf{Merkmale Team vs. Gruppe}\\
            \ra Multipersonalität: mind. 3 MA\\
            \ra gegenseitiger Abhängigkeit als Mittel zum Zweck\\
            \ra Zielorientierung
            \item \textbf{beschreibende Merkmale}\\
            \ra fachlich \& zwischenmenschliche Interaktionen\\
            \ra Erkennen von Mitgliedschaft in sozialer Einheit\\
            \ra strukturierte Beziehung: Rolle, Regel \& Norm\\
            \ra gegenseitige Beeinflussung: Einstellung \& Verhalten\\
            \ra individuelle Motivation durch persönliche Ziele
            \item Einzelkämpfergruppe: $\downarrow$ Sach- \&  Beziehungsebene
            \item Gesellige Gruppe: $\downarrow$ Sachebene
            \item Expertenteam: $\downarrow$ Beziehungsebene
            \item Echtes Team: Flex. \& \emph{never change a winning team}
        \end{itemize}
    \end{sectionbox}
    \begin{hintbox}{Groupthink - Erkennen Gruppendynamik}
        \begin{itemize}
            \item starker Fokus auf Harmonie anstatt auf Produktivität
            \item wenig konstruktive Auseinandersetzungen\\
            \item $\uparrow$ Loyalität \& $\uparrow$ Solidarität, d.h. $\uparrow$ Teamfrieden
            \item Gegensteuerung der FK durch Förderung\\
            \ra kritische fachliche Diskussionen zum Arbeitsprozess\\
            \ra offener Umgang mit fachlichen Konflikten\\
            \ra gelegentliche Änderung  die Teamzusammensetzung
        \end{itemize}
    \end{hintbox}
    \begin{sectionbox}{Chancen \& Risiken der Teamarbeit}
        \begin{itemize}
            \item $\uparrow$ Entscheidungsqualität Multipersonalität: Groupthink
            \item $\uparrow$ Effizienz diverse Stärken: $\uparrow$ Koordinationsaufwand
            \item Bündelung von Kompetenzen: $\uparrow$ Konfliktgefahr
            \item Risikoverringerung durch Kommunikation \& Kontrolle: Subkultur im Team \& Abgrenzung Unternehmen
        \end{itemize}
    \end{sectionbox}
    \begin{sectionbox}{Lebenszyklus von Teams}
        \begin{itemize}
            \item Entwicklung in Phasen: unterschiedliche FK Aufgaben
            \item \textbf{Forming} (\textbf{\emph{Beziehungsebene}}): \textbf{Beziehungsmanager}\\
            \ra Kennenlernen des Teams\\
            \ra Identifikation von Stärken \& Schwächen
            \item \textbf{Storming} (\textbf{\emph{Beziehungsebene}}):  \textbf{Schlichter}\\
            \ra Einsatz MA gemäß Stärken \& Potenziale\\
            \ra Identifikation Autoritätspersonen\\
            \ra Deeskalation von Konflikten 
            \item \textbf{Norming} (\textbf{\emph{Sachebene}}): \textbf{Koordinator}\\
            \ra Vereinbarung von Zielen \& Aufgaben\\
            \ra Def. Leitlinien \& Kommunikationsprozessen
            \ra punktuelle Leistungskontrollen
            \item \textbf{Performing} (\textbf{\emph{Sach-/Beziehungsebene}}): \textbf{Coach}\\
            \ra Übertragen von Kompetenzen auf das Team\\
            \ra Unterstützen durch Strukturierungshilfen\\
            \ra Ressourcenausstattung
        \end{itemize}
    \end{sectionbox}
    \begin{sectionbox}{Shared Leadership = verteilte Führung}
        \begin{itemize}
            \item Gegenentwurf zur klassischer/hierarchischer Führung
            \item Führung als kollektiver \& sozialer Prozess
            \item MA übernehmen Teile der Führungsfunktion
            \item Verwechslungsgefahr: Führung von unten
        \end{itemize}
    \end{sectionbox}
    \begin{hintbox}{Voraussetzungen für Shared Leadership}
        \begin{itemize}
            \item geringe fachliche/kompetenzbasierte Überlegenheit FK
            \item demokratischer \& partizipativer Führungsstil
            \item Bereitschaft MA, Verantwortung zu übernehmen
            \item hohe Reife der MA
            \item positives Klima: Förderung Shared Leadership 
            \ra disjunkte Strategie bezüglich Führungsschwäche
        \end{itemize}
    \end{hintbox}
    \begin{sectionbox}{Organisationskultur als \emph{sozialer Klebstoff}}
        \begin{itemize}
            \item Führung im Kontext der Organisation\\
            \ra Kultur, Werte, gelebte Identität \& Image
            \item Konstituierende Elemente: Verhaltensweisen, Klima, Gruppennormen, Philosophie, Spielregeln \& Symbolik
            \item Drei-Ebenen-System nach \emph{Schein} \\
            \ra sichtbar, interpretationsbedürftig: Symbolik\\
            \ra teils sichtbar/unbewusst: Werte \& Standards\\
            \textbf{\emph{Kuluturkern}}\\
            \ra öffentlich, nach außen postuliert: proklamierte Basismaßnahmen, oft bei FK-Wechsel \emph{übergestülpt}
            \ra unsichtbar/unbewusst: Basismaßnahmen
        \end{itemize}
    \end{sectionbox}
    \begin{sectionbox}{Veränderungsmanagement}
        \begin{itemize}
            \item Strategische \& operative Umpositionierung \&  Umstrukturierung auf inhaltlicher \& Prozessebene 
            \item Folgen:Ablehnung der Veränderung durch MA\\
            \ra Kommunikation \& Transparenz
            \item Lösungsansätze: \emph{Mitnehmen} der MA\\
            \ra Identifizierung von Bedarfen für Veränderung\\
            \ra Sammeln von Ideen, Vorschläge für Anpassungen
        \end{itemize}
    \end{sectionbox}
    \begin{hintbox}{Vielfältige interne/externe Veränderungsprozesse }
        \begin{itemize}
            \item steigender Veränderungsdruck
            \item unsichere Situationen via Varianten \& Geografie
            \item Beschleunigung durch technologischen Fortschritt
            \item paradoxe Anforderungen wie Kooperation, Wettbewerb
            \item verstärktes Qualitätsbewusstsein in Märkten
            \item höhere Dienstleistungsansprüche der Kunden
        \end{itemize}
    \end{hintbox}
    \begin{sectionbox}{Achtstufiges Modell: Changemanagement}
        \begin{itemize}
            \item Dringlichkeit herstellen \& kommunizieren
            \item Koalitionen schmieden, die Veränderung einleiten
            \item strategische Vision entwickeln
            \item \emph{Armee von freiwilligen Unterstützern} aufbauen
            \item Hemmnisse \& Hindernisse abbauen
            \item \emph{Quick wins} anstreben
            \item Schwung beibehalten
            \item Veränderung verankern
        \end{itemize}
    \end{sectionbox}
